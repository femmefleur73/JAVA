1. 테이블스페이스 생성

ALTER SESSION SET "_ORACLE_SCRIPT"=TRUE;

/* 테이블스페이스 생성 */
CREATE TABLESPACE ORA_TS DATAFILE 'D:\oracle\ORADATA\ORA_TS.DBF' SIZE 100M AUTOEXTEND ON NEXT 50M MAXSIZE UNLIMITED; 

-- 유저생성
CREATE USER ORAUSER IDENTIFIED BY ORAUSER DEFAULT TABLESPACE ORA_TS TEMPORARY TABLESPACE TEMP QUOTA 500K ON ORA_TS;

-- user permission
GRANT CONNECT TO ORAUSER;
GRANT CONNECT, RESOURCE TO ORAUSER;
GRANT CONNECT, DBA TO ORAUSER;

-- 비밀번호 변경
ALTER USER ORAUSER IDENTIFIED BY "1qazxsw2";

 

2. 샘플 스키마 설치

* 명령 프롬프트에서 실행

cd /backup

imp ORAUSER/1qazxsw2 file=expall.dmp log=empall.log ignore=y grants=y rows=y indexes=y full=y
